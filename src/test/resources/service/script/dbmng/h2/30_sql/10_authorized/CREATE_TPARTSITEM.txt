CREATE TABLE IF NOT EXISTS TPARTSITEM(
	TPARTSITEM_ID INT NOT NULL AUTO_INCREMENT,
	ITEM_KEY VARCHAR(256),
	ITEM_QUERY VARCHAR(8192),
	TPARTSINPAGE_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TPARTSITEM_ID)
);
