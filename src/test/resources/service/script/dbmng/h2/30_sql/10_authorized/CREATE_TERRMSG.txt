CREATE TABLE IF NOT EXISTS TERRMSG(
	TERRMSG_ID INT NOT NULL AUTO_INCREMENT,
	SESSION_ID VARCHAR(256),
	TACCOUNT_ID INT,
	ERR_MSG VARCHAR(256),
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(TERRMSG_ID)
);
