CREATE TABLE IF NOT EXISTS MURIPATTERN(
	MURIPATTERN_ID INT NOT NULL AUTO_INCREMENT,
	URI_PATTERN_NAME VARCHAR(256),
	URI_PATTERN VARCHAR(256),
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(MURIPATTERN_ID)
);
