CREATE TABLE IF NOT EXISTS MATTR(
	MATTR_ID INT NOT NULL AUTO_INCREMENT,
	ATTR_NAME VARCHAR(256),
	ATTR_VAL VARCHAR(256),
	NUM_MIN INT,
	NUM_MAX INT,
	ATTR_NOTE VARCHAR(256),
	MATTRGRP_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(MATTR_ID)
);
