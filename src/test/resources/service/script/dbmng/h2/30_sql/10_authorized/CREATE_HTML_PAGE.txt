CREATE TABLE IF NOT EXISTS HTML_PAGE(
	HTML_PAGE_ID INT NOT NULL AUTO_INCREMENT,
	PAGE_NAME VARCHAR(256),
	URI_PATTERN_ID INT,
	SCR_ID_GET INT,
	RESP_KIND_GET VARCHAR(256),
	DESTINATION_GET VARCHAR(256),
	SCR_ID_POST INT,
	RESP_KIND_POST VARCHAR(256),
	DESTINATION_POST VARCHAR(256),
	SCR_ID_PUT INT,
	RESP_KIND_PUT VARCHAR(256),
	DESTINATION_PUT VARCHAR(256),
	SCR_ID_DELETE INT,
	RESP_KIND_DELETE VARCHAR(256),
	DESTINATION_DELETE VARCHAR(256),
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(HTML_PAGE_ID)
);
