CREATE TABLE IF NOT EXISTS MGNRKEYVAL(
	MGNRKEYVAL_ID INT NOT NULL AUTO_INCREMENT,
	GNR_KEY VARCHAR(256),
	GNR_VAL VARCHAR(256),
	MGNRGRP_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(MGNRKEYVAL_ID)
);
