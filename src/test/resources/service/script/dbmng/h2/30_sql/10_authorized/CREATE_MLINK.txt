CREATE TABLE IF NOT EXISTS MLINK(
	MLINK_ID INT NOT NULL AUTO_INCREMENT,
	LINK_NAME VARCHAR(256),
	MURIPATTERN_ID INT,
	IS_POST INT,
	MLINKGRP_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(MLINK_ID)
);
