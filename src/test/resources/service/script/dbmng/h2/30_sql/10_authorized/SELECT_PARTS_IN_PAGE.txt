	PARTS_IN_PAGE_ID,
	HTML_PAGE_ID,
	HTML_PARTS_ID,
	ORD_IN_GRP,
	VERSION,
	IS_DELETED,
	CREATED_BY,
	CREATED_AT,
	UPDATED_BY,
	UPDATED_AT
 FROM PARTS_IN_PAGE

	PARTS_IN_PAGE_ID,
	HTML_PAGE_ID,
	HTML_PARTS_ID,
	ORD_IN_GRP,
	VERSION,
	IS_DELETED,
	CREATED_BY,
	CREATED_AT,
	UPDATED_BY,
	UPDATED_AT
;
