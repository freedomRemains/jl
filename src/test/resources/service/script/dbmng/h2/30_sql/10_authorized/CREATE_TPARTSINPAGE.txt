CREATE TABLE IF NOT EXISTS TPARTSINPAGE(
	TPARTSINPAGE_ID INT NOT NULL AUTO_INCREMENT,
	MHTMLPAGE_ID INT,
	MHTMLPARTS_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TPARTSINPAGE_ID)
);
