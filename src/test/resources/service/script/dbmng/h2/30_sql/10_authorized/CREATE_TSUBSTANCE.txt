CREATE TABLE IF NOT EXISTS TSUBSTANCE(
	TSUBSTANCE_ID INT NOT NULL AUTO_INCREMENT,
	SUBSTANCE_NAME VARCHAR(256),
	SITE VARCHAR(256),
	STAT INT,
	DETAIL_DESC VARCHAR(8192),
	PRICE INT,
	VALID_FROM DATETIME,
	VALID_TO DATETIME,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TSUBSTANCE_ID)
);
