SELECT
	TSUBSTANCE_ID,
	SUBSTANCE_NAME,
	SITE,
	STAT,
	DETAIL_DESC,
	PRICE,
	VALID_FROM,
	VALID_TO,
	VERSION,
	IS_DELETED,
	CREATED_BY,
	CREATED_AT,
	UPDATED_BY,
	UPDATED_AT
 FROM TSUBSTANCE
 ORDER BY
	TSUBSTANCE_ID,
	SUBSTANCE_NAME,
	SITE,
	STAT,
	DETAIL_DESC,
	PRICE,
	VALID_FROM,
	VALID_TO,
	VERSION,
	IS_DELETED,
	CREATED_BY,
	CREATED_AT,
	UPDATED_BY,
	UPDATED_AT
;
