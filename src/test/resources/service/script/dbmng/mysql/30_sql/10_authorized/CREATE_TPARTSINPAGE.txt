CREATE TABLE IF NOT EXISTS TPARTSINPAGE(
	TPARTSINPAGE_ID INT NOT NULL AUTO_INCREMENT,
	MHTMLPAGE_ID INT,
	MHTMLPARTS_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(TPARTSINPAGE_ID)
);
