CREATE TABLE IF NOT EXISTS TERRMSG(
	TERRMSG_ID INT NOT NULL AUTO_INCREMENT,
	SESSION_ID VARCHAR(256),
	TACCOUNT_ID INT,
	ERR_MSG VARCHAR(256),
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TERRMSG_ID)
);
