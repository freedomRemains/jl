SELECT
	TBL_DEF_ID,
	TABLE_NAME,
	FIELD_NAME,
	TYPE_NAME,
	ALLOW_NULL,
	KEY_DIV,
	DEFAULT_VALUE,
	EXTRA,
	TABLE_LOGICAL_NAME,
	FIELD_LOGICAL_NAME,
	FOREIGN_TABLE,
	DESC_FIELD

	TBL_DEF_ID,
	TABLE_NAME,
	FIELD_NAME,
	TYPE_NAME,
	ALLOW_NULL,
	KEY_DIV,
	DEFAULT_VALUE,
	EXTRA,
	TABLE_LOGICAL_NAME,
	FIELD_LOGICAL_NAME,
	FOREIGN_TABLE,
	DESC_FIELD
;
