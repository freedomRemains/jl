CREATE TABLE IF NOT EXISTS TMAIL(
	TMAIL_ID INT NOT NULL AUTO_INCREMENT,
	MAIL_FROM VARCHAR(256),
	MAIL_SUBJECT VARCHAR(256),
	SENT_DATE DATETIME,
	HONBUN VARCHAR(8192),
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TMAIL_ID)
);
