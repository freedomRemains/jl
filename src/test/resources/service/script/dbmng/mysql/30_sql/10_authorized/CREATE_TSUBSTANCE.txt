CREATE TABLE IF NOT EXISTS TSUBSTANCE(
	TSUBSTANCE_ID INT NOT NULL AUTO_INCREMENT,
	SUBSTANCE_NAME VARCHAR(256),
	SITE VARCHAR(256),
	STAT INT,
	DETAIL_DESC VARCHAR(8192),
	PRICE INT,
	VALID_FROM DATETIME,
	VALID_TO DATETIME,
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(TSUBSTANCE_ID)
);
