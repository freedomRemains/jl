CREATE TABLE IF NOT EXISTS MHTMLPAGE(
	MHTMLPAGE_ID INT NOT NULL AUTO_INCREMENT,
	PAGE_NAME VARCHAR(256),
	MURIPATTERN_ID INT,
	TSCR_ID_GET INT,
	RESP_KIND_GET VARCHAR(256),
	DESTINATION_GET VARCHAR(256),
	TSCR_ID_POST INT,
	RESP_KIND_POST VARCHAR(256),
	DESTINATION_POST VARCHAR(256),
	TSCR_ID_PUT INT,
	RESP_KIND_PUT VARCHAR(256),
	DESTINATION_PUT VARCHAR(256),
	TSCR_ID_DELETE INT,
	RESP_KIND_DELETE VARCHAR(256),
	DESTINATION_DELETE VARCHAR(256),
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(MHTMLPAGE_ID)
);
