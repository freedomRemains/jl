CREATE TABLE IF NOT EXISTS MLINK(
	MLINK_ID INT NOT NULL AUTO_INCREMENT,
	LINK_NAME VARCHAR(256),
	MURIPATTERN_ID INT,
	IS_POST INT,
	MLINKGRP_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	DEL_FLG INT,
	CREATE_USER VARCHAR(128),
	CREATE_DATE DATETIME,
	UPDATE_USER VARCHAR(128),
	UPDATE_DATE TIMESTAMP,
	PRIMARY KEY(MLINK_ID)
);
