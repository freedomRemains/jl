CREATE TABLE IF NOT EXISTS TSCRELM(
	TSCRELM_ID INT NOT NULL AUTO_INCREMENT,
	SERVICE_NAME VARCHAR(256),
	ADAPTER VARCHAR(256),
	PREPARE_INPUT VARCHAR(256),
	TSCR_ID INT,
	ORD_IN_GRP INT,
	VERSION INT,
	IS_DELETED INT,
	CREATED_BY VARCHAR(128),
	CREATED_AT DATETIME,
	UPDATED_BY VARCHAR(128),
	UPDATED_AT TIMESTAMP,
	PRIMARY KEY(TSCRELM_ID)
);
